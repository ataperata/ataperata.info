NLP Sentiment Analysis


import nltk
from textblob import TextBlob
import re

def analyze_sentiment(text):
    # Clean the text
    cleaned_text = re.sub(r'[^a-zA-Z\s]', '', text)
    
    # Create TextBlob object
    blob = TextBlob(cleaned_text)
    
    # Get sentiment polarity
    polarity = blob.sentiment.polarity
    
    # Determine sentiment
    if polarity > 0.1:
        return "üòä Positive", polarity
    elif polarity < -0.1:
        return "üò¢ Negative", polarity
    else:
        return "üòê Neutral", polarity

# Sample texts to analyze
texts = [
    "I absolutely love this AI technology!",
    "This is terrible and disappointing.",
    "The weather is okay today."
]

for text in texts:
    sentiment, score = analyze_sentiment(text)
    print(f"Text: {text}")
    print(f"Sentiment: {sentiment} (Score: {score:.2f})")
    print("-" * 50)

    def create_quantum_circuit(self):
        # Create quantum and classical registers
        qreg = QuantumRegister(self.quantum_bits, 'q')
        creg = ClassicalRegister(self.quantum_bits, 'c')
        circuit = QuantumCircuit(qreg, creg)
        
        # Apply quantum gates for AI processing
        for i in range(self.quantum_bits):
            circuit.h(qreg[i])  # Hadamard gate for superposition
            if i > 0:
                circuit.cx(qreg[i-1], qreg[i])  # Entanglement
        
        # Apply rotation gates (simulating AI learning)
        for i in range(0, self.quantum_bits, 2):
